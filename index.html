// ============================================ // FUNÇÃO PARA REGISTRAR SAÍDA (ATUALIZADA) // ============================================ function registrarSaida(event) { event.preventDefault(); const bobinaId = parseInt(document.getElementById('sBobinaId').value); const equipe = document.getElementById('sEquipe').value; const funcionario = document.getElementById('sFuncionario').value; const ordem = document.getElementById('sOrdem').value; const metros = parseFloat(document.getElementById('sMetros').value); const obs = document.getElementById('sObservacao').value; if (!bobinaId) { alert('❌ Selecione uma bobina!'); return; } const bobina = bobinas.find(b => b.id === bobinaId); if (!bobina) { alert('❌ Bobina não encontrada!'); return; } const disponivel = bobina.metragemInicial - bobina.metrosUsados; if (metros > disponivel) { alert(`❌ Disponível: ${disponivel}m`); return; } if (metros <= 0) { alert('❌ Informe uma quantidade válida!'); return; } // Atualizar bobina bobina.metrosUsados += metros; const restante = bobina.metragemInicial - bobina.metrosUsados; if (restante === 0) bobina.status = 'Finalizada'; else if (bobina.metrosUsados > 0) bobina.status = 'Em uso'; // Registrar movimentação const now = new Date(); movimentacoes.push({ id: movimentacoes.length + 1, bobinaId: bobina.id, equipe: equipe, funcionario: funcionario, ordemServico: ordem, metrosUtilizados: metros, observacao: obs, data: now.toLocaleDateString('pt-BR'), hora: now.toLocaleTimeString('pt-BR', { hour: '2-digit', minute: '2-digit' }) }); alert(`✅ Saída registrada!\n📏 ${metros}m utilizados\n📦 Restante: ${restante}m`); // Limpar formulário document.getElementById('sBobinaId').value = ''; document.getElementById('sEquipe').value = ''; document.getElementById('sFuncionario').value = ''; document.getElementById('sOrdem').value = ''; document.getElementById('sMetros').value = ''; document.getElementById('sObservacao').value = ''; showSaida(); }
